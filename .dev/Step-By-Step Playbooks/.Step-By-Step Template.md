## Code Block 0 — Session setup + verify

Pattern: **prepare** (seat, table, tokens, dummy state) → **verify** (assert world ready) → HUMAN handoff.

```lua
-- Optional: hoist test constants for later blocks (session stays loaded).
local FIXTURE = {
  tableKey = "Table A",
  sceneSlotIndex = 3,
  sceneKey = "e2e_verify_stub",
  npcA = "myleneHamelin",
}

U.RunSequence({
  function()
    printHeader("Title: Session setup", 1)
  end,
  function()
    if #(Player.getPlayers() or {}) < 1 then
      error("[Setup FAIL] Host not connected")
    end
    rollE2eSeatPrep("Black")
    DEBUG.syncTableSimplified(FIXTURE.tableKey)
    if type(DEBUG.ensureSceneLibraryStub) ~= "function" then
      error("[Setup FAIL] DEBUG.ensureSceneLibraryStub missing — Save & Play after pulling latest repo Lua")
    end
    DEBUG.ensureSceneLibraryStub(FIXTURE.sceneSlotIndex, FIXTURE.sceneKey, {
      title = "E2E verify stub",
      tableKey = FIXTURE.tableKey,
    })
    DEBUG.spawnNpcControlBoardTokens({ destroyExisting = false })
    print("PASS — session prepared (seat, table, tokens, library stub)")
  end,
  function()
    printHeader("Title: Verify ready", 1)
  end,
  function()
    if not U.isStorytellerSteamPlayer("Black") then
      error("[Verify FAIL] Host not on Black after rollE2eSeatPrep")
    end
    if S.getStateVal("seatLayout", "currentTableKey") ~= FIXTURE.tableKey then
      error("[Verify FAIL] table key not " .. FIXTURE.tableKey)
    end
    print("PASS — prerequisites satisfied")
  end,
  function()
    print("   ▶▶▶ HUMAN ▶▶▶ Run Code Block A.1.")
  end,
})
```

    TITLE.helperFunctionOne = function()
      return true
    end
    TITLE.helperFunctionTwo = function()
      return true
    end
    TITLE.helperFunctionThree = function()
      return true
    end
  end,
  function()
    print("Helper functions initialized.")
  end,
  function()
    printHeader("PHASE 2.1 — First test of dummyFunction_X", 1)
  end,
  function()
    local param1, param2 = "alpha", 42
    print(">> Calling dummyFunction_X(" .. tostring(param1) .. ", " .. tostring(param2) .. ") ...")
  end,
  function()
    local param1, param2 = "alpha", 42
    local logLines = {}
    local ok, result = xpcall(
      function()
        return dummyFunction_X(param1, param2)
      end,
      function(err)
        return tostring(err) .. "\n" .. debug.traceback("", 2)
      end
    )
    if not ok then
      table.insert(logLines, "")
      table.insert(logLines, "FAIL — dummyFunction_X raised an error")
      table.insert(logLines, "   " .. tostring(result))
      table.insert(logLines, "")
      print(table.concat(logLines, "\n"))
      error("[Phase 2.1 FAIL] dummyFunction_X")
    end
    -- Real playbooks: prefer rollConfirm({ ... }) or domain DEBUG asserts instead of custom PASS lines.
    table.insert(logLines, "PASS — dummyFunction_X returned " .. tostring(result))
    print(table.concat(logLines, "\n"))
  end,
  function()
    -- LAST RESORT: subjective visual when Lua cannot assert (lighting, timing, hand-zone pose).
    -- Prefer a Lua assert in Code Block A.2 when feasible.
    print("   ▶▶▶ HUMAN ▶▶▶ Visually confirm the Purple player's hand zone is correctly positioned and rotated, then run Code Block A.2.")
